```
type People {
	name: str
	favorite str
	age,weight,height: uint
	( name ) { ... }
	( name []str ) { ... }
	( name avgRatio: uint ) { ... }
}

// 直接量

People{ 'Jackie' 12 35 } // 按照定义的顺序初始化属性
People{ age: 34 name: 'Jackie' } // 指定属性名进行初始化
let weight = 42
People{: age: 34 weight } // 使用解构进行初始化
People{: weight } // 使用解构进行初始化

// 通过构造函数进行创建
People( 'Boom' 34 ) // 参数序列
let avgRatio = 42
People( name: 'Boom' avgRatio: 30 ) // 指定描述
People(: name: 'Boom' avgRatio ) // 指定描述
People($ Boom basketball, football, paint ) // 当所有形式参数都为字符串时，可以使用模板化定义

// 类型定义

People p1 = { 'Jackie' }
People p2 = ( 'Jackie' )

People p3, p4, p5
p3 = { 'Jackie' }
p4 = ( 'Jackie' )

type Worker like People {
	work() {}
}
p5 = Worker{ 'Jackie' }
```

###### 混合式数据传递规则

+ 提取所有`指定描述`并忽略
+ 提取所有定义中已被`指定描述`的属性项并忽略
+ 使剩余`实体数据`对应剩余`定义属性`
