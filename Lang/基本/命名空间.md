```
type A {
    name: str
    age: uint
}

type B {
    name: str
    weight: float
}

type C like A {
    ( n: str ) {
        name = n
    }
}

type E like A {
    - A.name
    ( name ) {
        name = n
    }
}

type D like A,B {
    ( n: str ) {
        // 当组合
        A.name = n
        B.name = n.toUpperCast()
    }
}
```
