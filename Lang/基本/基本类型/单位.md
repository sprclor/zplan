###### 单位簇

```
unit Distance {
	km 	1e3		// 千米
	m	1e0		// 米
	dm	1e-1	// 分米
	cm 	1e-2	// 厘米
	mm	1e-3	// 毫米
	um	1e-6	// 微米
	nm	1e-9	// 纳米
	pm	1e-12	// 皮米
	fm	1e-15	// 飞米
}

unit Area = Distance * Distance {
	km2 km
	m2	m
	dm2 dm
	cm2	cm
	mm2 mm
	um2	um
	pm2	pm
	fm2 fm
}

let a = 1.423km
type.( a )			// Distance
type.( a * a )			// Area
m.( a )				// 1423
fm.( a )				// 1,423,000,000,000,000,000

fn getArea( d1, d2: Distance ) Area {
	return d1 * d2
}

getArea( 1.2cm 2.4m ) // 0.0288m2
```
