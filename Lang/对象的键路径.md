```
m := {
    a: {
        b: {
            c: {
                d: {
                    v: 1,
                    v2: [1,2,3,4],
                }
            }
        },
        z: {
            y: 1
        }
    }
}

m[a/b/c/d]	// { v: 1, v2: [1,2,3,4] }
m[a/z/y]	// 1
m[a/x]		// nil

skey := "b"
m[a/@skey/c/d/v2]	// [1,2,3,4]

m[a/skey/c/d/v2]

skey = "z"
m[a/@skey/c/d/v2]	// nil
```
