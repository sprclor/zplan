`白描体`是一种强调实体结构的一种复合类型。

与白描体相关的话题在系统中有两种，分别是`白描体结构实体`和`白描体素体`。

白描体结构实体是一种模板式的、抽象的实体，通过具体化来产生白描体素体。

```
// 定义白描体结构
stroke {
	a {
		b {
			c {
				...
			}
		}
	}
}
```

它可以被用来表达实体的逻辑位置。

白描体的结构很简单



```

type View {}
type Slice like View {}

type StackView {
	spacing Distance
	stroke {
		top {
			> top: View
		}
		bottom {
			> : View
		}
		status {
			Side
		}
	}
	stroke Side {
		main {
			> top
		}
		other {
			>
		}
	}
	
	$( > top ) {
		padding 2em
	}
}

sv := StackView {
	4em
	
	< .top Slice { }
	< Side.top View { }
	< Side View { }
}

$sv ( > top ) { padding 1em }
$( sv.Side > main ) { }
```

```
type Item {
	img		Image
	desc	str
	type	Category
    stroke {
        img
        desc {
        	:Tx &desc
        }
    }
    
    $ {
    	$ > img {
    		ratio: 1.2
    	}
    	$ > desc {
    		padding: 0 1em
    	}
    }
}

type Inventory {
	items []Item
	
	filter( ) []Item { ... }
	
    stroke {
        @for filter(items) as item {
            item
        }
    }
}

inventory := Inventory {
	Item {
		{\ Jackie's\ sword }
		"Jackie\'s sword."
		Category.Equipment.Weapon.Sword
	}
	Item {
		{\ Brick }
		"This is a brick."
		Category.( Equipment.Weapon Materials Building )
	}
}
```
